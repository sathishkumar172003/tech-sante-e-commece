@app.route('/searchResult', methods=["POST", "GET"]) def search_result(): if request.method == "POST": search_word = request.form.get('search') products = Product.query.msearch(search_word, fields=['name', 'description', 'price'], limit=3) return render_template('products/all_products_page.html', products = products) return render_template('products/search_result.html')
